# 大模型收藏接口文档

本文档说明如何在第三方 HTML 页面中，通过 Web Crypto API 解密远程加密的 `model.db.enc` 数据库文件，弹窗选择已收藏的大模型。

---

## 1. 文件地址与格式

| 项目 | 说明 |
|---|---|
| **远程 URL** | `https://web.tuchong.tk/大模型API调试与管理工具/model.db.enc` |
| **数据库引擎** | SQLite（通过浏览器端 [sql.js](https://github.com/sql-js/sql.js) 读取） |
| **加密算法** | PBKDF2-SHA256（60 万次迭代）派生 256-bit AES-GCM 密钥 |
| **加密文件结构** | salt(16字节) + nonce(12字节) + ciphertext(含GCM tag，末尾16字节) |
| **表结构** | `favorite_models(id, apiId, platform, url, key, modelName, remark)` |

---

## 2. 依赖引入

```html
<script src="https://cdn.jsdelivr.net/npm/sql.js@1.10.3/dist/sql-wasm.js"></script>
```

sql.js 会在运行时通过 WebAssembly 加载 SQLite 引擎，`locateFile` 配置指向 CDN 路径（见下文代码）。

## 3. 解密流程

```
用户输入密码
  → fetch 远程 model.db.enc（ArrayBuffer）
  → 拆解 salt(0-15) + nonce(16-27) + ciphertext(28往后)
  → PBKDF2-SHA256(salt, 600000次) 派生 AES-256 密钥
  → AES-GCM 解密（nonce 作为 iv）
  → 得到 SQLite 字节 → sql.js 解析 → 提取平台和模型列表
```

### 核心解密函数

```javascript
// 派生密钥
async function deriveKey(password, salt, usage) {
  const enc = new TextEncoder();
  const pwKey = await crypto.subtle.importKey('raw', enc.encode(password), 'PBKDF2', false, ['deriveKey']);
  return crypto.subtle.deriveKey(
    { name: 'PBKDF2', salt, iterations: 600000, hash: 'SHA-256' },
    pwKey,
    { name: 'AES-GCM', length: 256 },
    false,
    [usage]
  );
}

// 解密（接收 Uint8Array + 密码，返回解密后的 Uint8Array）
async function decryptBytes(data, password) {
  const salt = data.slice(0, 16);
  const nonce = data.slice(16, 28);
  const ciphertext = data.slice(28);
  const key = await deriveKey(password, salt, 'decrypt');
  return new Uint8Array(await crypto.subtle.decrypt({ name: 'AES-GCM', iv: nonce }, key, ciphertext));
}

// 完整流程：fetch → 解密 → sql.js 解析
async function loadModels(password) {
  const SQL = await initSqlJs({
    locateFile: f => `https://cdn.jsdelivr.net/npm/sql.js@1.10.3/dist/${f}`
  });

  const resp = await fetch('https://web.tuchong.tk/大模型API调试与管理工具/model.db.enc');
  if (!resp.ok) throw new Error('无法加载数据库 (HTTP ' + resp.status + ')');

  const encrypted = new Uint8Array(await resp.arrayBuffer());
  const decrypted = await decryptBytes(encrypted, password);

  const db = new SQL.Database(decrypted);
  const result = db.exec('SELECT * FROM favorite_models ORDER BY platform, modelName');
  db.close();

  if (!result.length) throw new Error('数据库中没有模型数据');

  return result[0].values.map(row => ({
    id: row[0],
    apiId: row[1],
    platform: row[2] || '',
    url: row[3] || '',
    key: row[4] || '',
    modelName: row[5] || '',
    remark: row[6] || ''
  }));
}
```

### JavaScript 逻辑

```javascript
var models = [];
var overlay = document.getElementById('settingsModal');

// ===== 存储相关 =====

function saveManual() {
  var cfg = {
    baseUrl: document.getElementById('apiBaseUrl').value.trim(),
    apiKey: document.getElementById('apiKey').value.trim(),
    model: document.getElementById('apiModel').value.trim()
  };
  if (!cfg.baseUrl) return showError('请填写 API Base URL');
  if (!cfg.apiKey) return showError('请填写 API Key');
  if (!cfg.model) return showError('请填写模型名称');
  sessionStorage.setItem('my_tool_config', JSON.stringify(cfg));
  sessionStorage.setItem('my_tool_manual', JSON.stringify(cfg));
  overlay.classList.remove('active');
}

function clearAll() {
  sessionStorage.removeItem('my_tool_config');
  sessionStorage.removeItem('my_tool_manual');
  document.getElementById('apiBaseUrl').value = '';
  document.getElementById('apiKey').value = '';
  document.getElementById('apiModel').value = '';
}

// ===== 弹框入口 =====

document.getElementById('openSettingsBtn').addEventListener('click', openSettings);
document.getElementById('saveBtn').addEventListener('click', saveManual);
document.getElementById('clearBtn').addEventListener('click', clearAll);
overlay.addEventListener('click', function(e) {
  if (e.target === overlay) overlay.classList.remove('active');
});

function openSettings() {
  // 强制清空三个输入框（防浏览器自动填充）
  document.getElementById('apiBaseUrl').value = '';
  document.getElementById('apiKey').value = '';
  document.getElementById('apiModel').value = '';
  // 从 manual 存储恢复（仅手动保存过的值会回显）
  try {
    var saved = sessionStorage.getItem('my_tool_manual');
    if (saved) {
      var cfg = JSON.parse(saved);
      document.getElementById('apiBaseUrl').value = cfg.baseUrl || '';
      document.getElementById('apiKey').value = cfg.apiKey || '';
      document.getElementById('apiModel').value = cfg.model || '';
    }
  } catch(e) {}

  overlay.classList.add('active');

  // 检查解密缓存
  var cached = sessionStorage.getItem('decrypted_models');
  if (cached) {
    models = JSON.parse(cached);
    renderModelListView();
  } else {
    renderPasswordView();
  }
}

// ===== 密码视图 =====

function renderPasswordView() {
  document.getElementById('pickerContent').innerHTML =
    '<div class="field">' +
      '<label>解密密码</label>' +
      '<input type="password" id="pwInput" placeholder="请输入数据库解密密码" autocomplete="off">' +
    '</div>' +
    '<p id="errorMsg" class="error" style="display:none"></p>' +
    '<button class="btn btn-primary" id="decryptBtn">🔓 解密并加载模型列表</button>';

  document.getElementById('decryptBtn').addEventListener('click', doDecrypt);
  document.getElementById('pwInput').addEventListener('keyup', function(e) {
    if (e.key === 'Enter') doDecrypt();
  });
  document.getElementById('pwInput').focus();
}

async function doDecrypt() {
  var pw = document.getElementById('pwInput').value.trim();
  var errEl = document.getElementById('errorMsg');
  var btn = document.getElementById('decryptBtn');
  if (!pw) { errEl.textContent = '请输入密码'; errEl.style.display = 'block'; return; }
  btn.disabled = true; btn.textContent = '解密中...'; errEl.style.display = 'none';
  try {
    models = await loadModels(pw);
    sessionStorage.setItem('decrypted_models', JSON.stringify(models));
    renderModelListView();
  } catch (e) {
    if (e.name === 'OperationError') {
      errEl.textContent = '密码错误，请重新输入';
    } else {
      errEl.textContent = '解密失败: ' + e.message;
    }
    errEl.style.display = 'block';
    btn.disabled = false; btn.textContent = '🔓 解密并加载模型列表';
  }
}

// ===== 模型列表视图 =====

function renderModelListView() {
  var platforms = [...new Set(models.map(function(m) { return m.platform; }))].sort();
  var html =
    '<div class="field"><label>平台</label><select id="platSelect">' +
    platforms.map(function(p) {
      return '<option value="' + escAttr(p) + '">' + escHtml(p) + '</option>';
    }).join('') +
    '</select></div>' +
    '<div class="field"><label>模型</label><select id="modelSelect"></select></div>' +
    '<p id="confirmError" class="error" style="display:none"></p>' +
    '<button class="btn btn-success" id="confirmBtn">✅ 确认使用</button>';

  document.getElementById('pickerContent').innerHTML = html;

  var platSelect = document.getElementById('platSelect');
  var modelSelect = document.getElementById('modelSelect');

  function filterModels() {
    var plat = platSelect.value;
    var filtered = models.filter(function(m) { return m.platform === plat; });
    modelSelect.innerHTML = filtered.map(function(m) {
      return '<option value="' + escAttr(m.modelName) + '">' + escHtml(m.modelName) + '</option>';
    }).join('');
  }

  platSelect.addEventListener('change', filterModels);

  // 从 localStorage 恢复上次选中的平台和模型
  var lastPick = {};
  try { lastPick = JSON.parse(localStorage.getItem('last_pick') || '{}'); } catch (e) {}

  if (lastPick.platform && platforms.indexOf(lastPick.platform) !== -1) {
    platSelect.value = lastPick.platform;
  }
  filterModels();

  if (lastPick.modelName) {
    var idx = -1;
    for (var i = 0; i < modelSelect.options.length; i++) {
      if (modelSelect.options[i].value === lastPick.modelName) {
        idx = i; break;
      }
    }
    if (idx !== -1) modelSelect.selectedIndex = idx;
  }

  document.getElementById('confirmBtn').addEventListener('click', function() {
    var plat = platSelect.value;
    var modelName = modelSelect.value;
    var m = models.find(function(x) { return x.platform === plat && x.modelName === modelName; });
    if (!m) return;

    // 记住本次选择（永久存储）
    localStorage.setItem('last_pick', JSON.stringify({ platform: plat, modelName: modelName }));

    // 仅写入 sessionStorage['my_tool_config']，不写入 manual key
    // 因此重新打开弹框时输入框不会回显收藏库的值
    var cfg = { baseUrl: m.url, apiKey: m.key, model: m.modelName };
    sessionStorage.setItem('my_tool_config', JSON.stringify(cfg));

    overlay.classList.remove('active');
  });
}

// ===== 辅助函数 =====

function escHtml(str) {
  if (str == null) return '';
  return String(str).replace(/&/g, '&amp;').replace(/</g, '&lt;').replace(/>/g, '&gt;');
}
function escAttr(str) {
  if (str == null) return '';
  return String(str).replace(/&/g, '&amp;').replace(/"/g, '&quot;').replace(/</g, '&lt;').replace(/>/g, '&gt;');
}
function showError(msg) {
  alert(msg); // 简化演示，实际项目可使用 toast
}
```

## 5. 数据存储设计

| 存储键 | 存储位置 | 写入场景 | 生命周期 |
|--------|---------|---------|---------|
| `my_tool_config` | `sessionStorage` | 手动保存 + 收藏库确认 | 会话级（标签页关闭即清除） |
| `my_tool_manual` | `sessionStorage` | 仅手动保存按钮 | 会话级 |
| `last_pick` | `localStorage` | 收藏库确认时 | 永久（跨会话） |
| `decrypted_models` | `sessionStorage` | 解密成功时 | 会话级 |

**说明**：

- **`my_tool_config`** — AI 生成/API 调用时读取的配置数据源。优先使用 `sessionStorage`，回退到 DOM 输入框当前值。
- **`my_tool_manual`** — 仅由「保存」按钮写入，仅用于弹框打开时回显到输入框。收藏库确认不会写入此 key，因此在两种输入方式之间做了严格隔离。
- **`last_pick`** — 用户从收藏库选中的平台和模型名称，用于下次打开弹框时自动恢复下拉列表的选中状态。
- **`decrypted_models`** — 解密后的模型列表缓存，同一标签页内再次打开弹框无需重复解密。

## 6. 浏览器自动填充防范

浏览器对 `type="password"` 输入框有强自动填充行为，`autocomplete="off"` 多数浏览器已不生效。本方案采用二层防御：

```html
<!-- 必须使用 autocomplete="new-password"，不可用 "off" -->
<input type="password" id="apiKey" autocomplete="new-password">
```

```javascript
// openSettings() 每次打开弹框时先强制清空 DOM，再从 manual key 恢复
// 这样浏览器填充的值会被清掉，回显的只有之前手动保存过的值
document.getElementById('apiBaseUrl').value = '';
document.getElementById('apiKey').value = '';
document.getElementById('apiModel').value = '';
// 然后从 sessionStorage['my_tool_manual'] 恢复
```

## 7. 数据模型

`loadModels` 返回的每个模型对象字段：

| 字段 | 类型 | 说明 |
|---|---|---|
| `id` | number | 收藏记录 ID |
| `apiId` | number | 关联的 API 配置 ID |
| `platform` | string | 平台名称（代理平台带"（反代）"后缀） |
| `url` | string | API 基础地址（代理平台含 `http://localhost:8317/proxy/` 前缀） |
| `key` | string | API Key |
| `modelName` | string | 模型名称 |
| `remark` | string | 备注 |

## 8. 注意事项

1. **sql.js WASM 加载**：必须通过 `locateFile` 配置 CDN 路径，否则浏览器无法加载 `.wasm` 文件。
2. **CORS 限制**：`model.db.enc` 部署在 GitHub Pages 时没有 CORS 头，浏览器直接 `fetch` 可能被拦截。解决方案：
   - 将文件部署在无 CORS 限制的位置
   - 或通过自己的后端代理转发
3. **密码仅存内存**：解密密码不持久化，刷新页面需重新输入。
4. **加密格式兼容**：salt(16) + nonce(12) + ciphertext 结构，与主项目「大模型 API 调试与管理工具」的导出格式完全一致。
5. **表结构不可变**：`model.db.enc` 由主项目生成导出，集成方只读使用，无需关心写入逻辑。
6. **存储 key 名称**：本文档示例使用 `my_tool_config` / `my_tool_manual` / `last_pick`，与 `index.html` 中使用的 `wechat_editor_config` / `wechat_editor_manual` / `wechat_editor_last_pick` 不同。集成时请根据实际项目统一命名，避免 `sessionStorage` 冲突。
