# 大模型收藏接口文档

本文档说明如何在第三方 HTML 页面中，通过 Web Crypto API 解密远程加密的 `model.db.enc` 数据库文件，弹窗选择已收藏的大模型。

---

## 1. 文件地址与格式

| 项目 | 说明 |
|---|---|
| **远程 URL** | `https://web.tuchong.tk/大模型API调试与管理工具/model.db.enc` |
| **数据库引擎** | SQLite（通过浏览器端 [sql.js](https://github.com/sql-js/sql.js) 读取） |
| **加密算法** | PBKDF2-SHA256（60 万次迭代）派生 256-bit AES-GCM 密钥 |
| **加密文件结构** | salt(16字节) + nonce(12字节) + ciphertext(含GCM tag，末尾16字节) |
| **表结构** | `favorite_models(id, apiId, platform, url, key, modelName, remark)` |

---

## 2. 依赖引入

```html
<script src="https://cdn.jsdelivr.net/npm/sql.js@1.10.3/dist/sql-wasm.js"></script>
```

sql.js 会在运行时通过 WebAssembly 加载 SQLite 引擎，`locateFile` 配置指向 CDN 路径（见下文代码）。

## 3. 解密流程

```
用户输入密码
  → fetch 远程 model.db.enc（ArrayBuffer）
  → 拆解 salt(0-15) + nonce(16-27) + ciphertext(28往后)
  → PBKDF2-SHA256(salt, 600000次) 派生 AES-256 密钥
  → AES-GCM 解密（nonce 作为 iv）
  → 得到 SQLite 字节 → sql.js 解析 → 提取平台和模型列表
```

### 核心解密函数

```javascript
// 派生密钥
async function deriveKey(password, salt, usage) {
  const enc = new TextEncoder();
  const pwKey = await crypto.subtle.importKey('raw', enc.encode(password), 'PBKDF2', false, ['deriveKey']);
  return crypto.subtle.deriveKey(
    { name: 'PBKDF2', salt, iterations: 600000, hash: 'SHA-256' },
    pwKey,
    { name: 'AES-GCM', length: 256 },
    false,
    [usage]
  );
}

// 解密（接收 Uint8Array + 密码，返回解密后的 Uint8Array）
async function decryptBytes(data, password) {
  const salt = data.slice(0, 16);
  const nonce = data.slice(16, 28);
  const ciphertext = data.slice(28);
  const key = await deriveKey(password, salt, 'decrypt');
  return new Uint8Array(await crypto.subtle.decrypt({ name: 'AES-GCM', iv: nonce }, key, ciphertext));
}

// 完整流程：fetch → 解密 → sql.js 解析
async function loadModels(password) {
  const SQL = await initSqlJs({
    locateFile: f => `https://cdn.jsdelivr.net/npm/sql.js@1.10.3/dist/${f}`
  });

  const resp = await fetch('https://web.tuchong.tk/大模型API调试与管理工具/model.db.enc');
  if (!resp.ok) throw new Error('无法加载数据库 (HTTP ' + resp.status + ')');

  const encrypted = new Uint8Array(await resp.arrayBuffer());
  const decrypted = await decryptBytes(encrypted, password);

  const db = new SQL.Database(decrypted);
  const result = db.exec('SELECT * FROM favorite_models ORDER BY platform, modelName');
  db.close();

  if (!result.length) throw new Error('数据库中没有模型数据');

  return result[0].values.map(row => ({
    id: row[0],
    apiId: row[1],
    platform: row[2] || '',
    url: row[3] || '',
    key: row[4] || '',
    modelName: row[5] || '',
    remark: row[6] || ''
  }));
}
```

### JavaScript 逻辑

```javascript
let models = [];       // 所有模型数据
let decrypted = false; // 是否已解密
const overlay = document.getElementById('pickerOverlay');
const content = document.getElementById('pickerContent');

// 打开弹框
document.getElementById('openPicker').addEventListener('click', () => {
  overlay.classList.add('active');
  if (!decrypted) {
    renderPasswordView();
  } else {
    renderModelListView();
  }
});

// 关闭弹框（点击遮罩层）
overlay.addEventListener('click', (e) => {
  if (e.target === overlay) overlay.classList.remove('active');
});

// ----- 密码视图 -----
function renderPasswordView() {
  content.innerHTML = `
    <div class="field">
      <label>解密密码</label>
      <input type="password" id="pwInput" placeholder="请输入数据库解密密码" autocomplete="off">
    </div>
    <p id="errorMsg" class="error" style="display:none"></p>
    <button class="btn btn-primary" id="decryptBtn">🔓 解密并加载模型列表</button>
  `;

  document.getElementById('decryptBtn').addEventListener('click', doDecrypt);
  document.getElementById('pwInput').addEventListener('keyup', (e) => {
    if (e.key === 'Enter') doDecrypt();
  });
  document.getElementById('pwInput').focus();
}

async function doDecrypt() {
  const pw = document.getElementById('pwInput').value.trim();
  const err = document.getElementById('errorMsg');
  const btn = document.getElementById('decryptBtn');
  if (!pw) { err.textContent = '请输入密码'; err.style.display = 'block'; return; }

  btn.disabled = true; btn.textContent = '解密中...';
  err.style.display = 'none';

  try {
    models = await loadModels(pw);
    decrypted = true;
    renderModelListView();
  } catch (e) {
    if (e.name === 'OperationError' || /password|密码/i.test(e.message)) {
      err.textContent = '密码错误，请重新输入';
    } else {
      err.textContent = '解密失败: ' + e.message;
    }
    err.style.display = 'block';
    btn.disabled = false; btn.textContent = '🔓 解密并加载模型列表';
  }
}

// ----- 模型列表视图 -----
function renderModelListView() {
  // 提取平台列表（去重排序）
  const platforms = [...new Set(models.map(m => m.platform))].sort();
  const firstPlat = platforms[0] || '';
  const firstModels = models.filter(m => m.platform === firstPlat);
  const first = firstModels[0] || {};

  content.innerHTML = `
    <div class="field">
      <label>平台</label>
      <select id="platSelect">
        ${platforms.map(p => `<option value="${escAttr(p)}">${escHtml(p)}</option>`).join('')}
      </select>
    </div>
    <div class="field">
      <label>模型</label>
      <select id="modelSelect">
        ${firstModels.map(m => `<option value="${escAttr(m.modelName)}">${escHtml(m.modelName)}</option>`).join('')}
      </select>
    </div>
    <div class="divider">或自定义</div>
    <div class="field">
      <label>Base URL</label>
      <input type="text" id="customUrl" placeholder="https://api.openai.com/v1" value="${escAttr(first.url || '')}">
    </div>
    <div class="field">
      <label>API Key</label>
      <input type="password" id="customKey" placeholder="sk-..." value="${escAttr(first.key || '')}">
    </div>
    <div class="field">
      <label>Model（模型名称）</label>
      <input type="text" id="customModel" placeholder="gpt-4o" value="${escAttr(first.modelName || '')}">
    </div>
    <p id="confirmError" class="error" style="display:none"></p>
    <button class="btn btn-success" id="confirmBtn">✅ 确认使用此模型</button>
  `;

  // 平台切换 → 过滤模型
  document.getElementById('platSelect').addEventListener('change', () => {
    const plat = document.getElementById('platSelect').value;
    const filtered = models.filter(m => m.platform === plat);
    const sel = document.getElementById('modelSelect');
    sel.innerHTML = filtered.map(m =>
      `<option value="${escAttr(m.modelName)}">${escHtml(m.modelName)}</option>`
    ).join('');
    if (filtered.length) {
      document.getElementById('customUrl').value = filtered[0].url || '';
      document.getElementById('customKey').value = filtered[0].key || '';
      document.getElementById('customModel').value = filtered[0].modelName || '';
    }
  });

  // 模型切换 → 填充自定义输入框
  document.getElementById('modelSelect').addEventListener('change', () => {
    const plat = document.getElementById('platSelect').value;
    const modelName = document.getElementById('modelSelect').value;
    const m = models.find(x => x.platform === plat && x.modelName === modelName);
    if (m) {
      document.getElementById('customUrl').value = m.url || '';
      document.getElementById('customKey').value = m.key || '';
      document.getElementById('customModel').value = m.modelName || '';
    }
  });

  // 确认按钮
  document.getElementById('confirmBtn').addEventListener('click', () => {
    const url = document.getElementById('customUrl').value.trim();
    const key = document.getElementById('customKey').value.trim();
    const model = document.getElementById('customModel').value.trim();
    const err2 = document.getElementById('confirmError');
    if (!url) { err2.textContent = '请填写 Base URL'; err2.style.display = 'block'; return; }
    if (!key) { err2.textContent = '请填写 API Key'; err2.style.display = 'block'; return; }
    if (!model) { err2.textContent = '请填写 Model 名称'; err2.style.display = 'block'; return; }

    // 输出结果到页面
    document.getElementById('resultUrl').textContent = url;
    document.getElementById('resultKey').textContent = key;
    document.getElementById('resultModel').textContent = model;
    overlay.classList.remove('active');
  });
}

// ----- 辅助函数 -----
function escHtml(str) {
  if (str == null) return '';
  return String(str).replace(/&/g, '&amp;').replace(/</g, '&lt;').replace(/>/g, '&gt;');
}
function escAttr(str) {
  if (str == null) return '';
  return String(str).replace(/&/g, '&amp;').replace(/"/g, '&quot;').replace(/</g, '&lt;').replace(/>/g, '&gt;');
}
```

## 5. 数据模型

`loadModels` 返回的每个模型对象字段：

| 字段 | 类型 | 说明 |
|---|---|---|
| `id` | number | 收藏记录 ID |
| `apiId` | number | 关联的 API 配置 ID |
| `platform` | string | 平台名称（代理平台带"（反代）"后缀） |
| `url` | string | API 基础地址（代理平台含 `http://localhost:8317/proxy/` 前缀） |
| `key` | string | API Key |
| `modelName` | string | 模型名称 |
| `remark` | string | 备注 |

---

## 6. 注意事项

1. **sql.js WASM 加载**：必须通过 `locateFile` 配置 CDN 路径，否则浏览器无法加载 `.wasm` 文件。
2. **CORS 限制**：`model.db.enc` 部署在 GitHub Pages 时没有 CORS 头，浏览器直接 `fetch` 可能被拦截。解决方案：
   - 将文件部署在无 CORS 限制的位置
   - 或通过自己的后端代理转发
3. **密码仅存内存**：解密密码不持久化，刷新页面需重新输入。
4. **加密格式兼容**：salt(16) + nonce(12) + ciphertext 结构，与主项目「大模型 API 调试与管理工具」的导出格式完全一致。
5. **表结构不可变**：`model.db.enc` 由主项目生成导出，第三方页面只读使用，无需关心写入逻辑。
